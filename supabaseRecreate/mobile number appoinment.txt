1. Add the column to appointments
ALTER TABLE appointments
ADD COLUMN patient_mobile TEXT;
2. Create trigger function to auto-fill it
CREATE OR REPLACE FUNCTION fill_patient_mobile()
RETURNS TRIGGER AS $$
BEGIN
  SELECT phone
  INTO NEW.patient_mobile
  FROM patients
  WHERE id = NEW.patient_id;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;
3. Attach trigger to appointments table
CREATE TRIGGER trg_fill_patient_mobile
BEFORE INSERT OR UPDATE ON appointments
FOR EACH ROW
EXECUTE FUNCTION fill_patient_mobile();
